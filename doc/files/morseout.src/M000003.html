def display_code(c)
  if(c == :dot)
    flash_led($sound_quanta)
    sleep($sound_quanta)
  elsif(c == :dash)
    flash_led($sound_quanta*3)
    sleep($sound_quanta)
  elsif(c == :letter_pause)
    sleep($sound_quanta*3)
  elsif(c == :word_pause)
    sleep($sound_quanta*7)
  else
    raise("unknown code #{c} passed to display_code")
  end
end
